async function loadNearbyPlaces(location, keyword = '', radius = 1500) {
  showLoading();
  clearMarkers();

  try {
    // Common params
    const openNowChecked = document.getElementById('open-now-checkbox').checked;
    const dessertMode = document.getElementById('dessert-checkbox').checked;

    // If dessertMode, do multi‐query textSearch
    let places = [];
    if (dessertMode) {
      const queries = [
        'dessert restaurant',
        'patisserie',
        'bakery',
        'chocolate shop',
        'ice cream',
        'cake shop'
      ];
      const seen = new Map();
      const service = new google.maps.places.PlacesService(map);

      // Helper to promisify textSearch
      const textSearch = opts => new Promise((res, rej) =>
        service.textSearch(opts, (r, s) => s === google.maps.places.PlacesServiceStatus.OK ? res(r) : rej(s))
      );

      for (let q of queries) {
        try {
          const results = await textSearch({
            query: q,
            location,
            radius,
            openNow: openNowChecked
          });
          for (let p of results) {
            if (!seen.has(p.place_id)) {
              seen.set(p.place_id, p);
            }
          }
        } catch (e) {
          console.warn('TextSearch error for', q, e);
        }
        if (seen.size >= 20) break; // enough
      }
      places = Array.from(seen.values());
    } else {
      // Normal nearby search
      const service = new google.maps.places.PlacesService(map);
      const nearby = opts => new Promise((res, rej) =>
        service.nearbySearch(opts, (r, s) => s === google.maps.places.PlacesServiceStatus.OK ? res(r) : rej(s))
      );
      places = await nearby({
        location,
        radius,
        type: keyword || currentPlaceType,
        openNow: openNowChecked
      });
    }

    // Now apply rating + distance + unwanted‐type filtering
    const filtered = filterAndSortPlaces(
      places,
      location,
      {
        radius,
        minRating: dessertMode ? 3.5 : (currentPlaceType === 'restaurant' ? 4.0 : 3.8),
        minReviews: currentPlaceType === 'lodging' ? 8 : (currentPlaceType === 'restaurant' ? 20 : 10),
        unwantedTypes: ["gas_station","car_repair","convenience_store","supermarket"]
      }
    );

    // Render
    const container = document.getElementById('places-container');
    container.innerHTML = '';
    if (!filtered.length) {
      container.innerHTML = `<div class="alert alert-info">No matches found in this area.</div>`;
    } else {
      filtered.forEach((place, i) => {
        const card = createPlaceCard(place, i);
        container.appendChild(card);
        addMarker(place, i);
      });
    }

  } catch (err) {
    console.error(err);
    document.getElementById('places-container').innerHTML =
      `<div class="alert alert-danger">Error loading places.</div>`;
  } finally {
    hideLoading();
  }
}
