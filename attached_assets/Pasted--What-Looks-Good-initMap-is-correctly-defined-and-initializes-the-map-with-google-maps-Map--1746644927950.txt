async function loadNearbyPlaces(location, keyword = '', radius = 1500) {
  showLoading();
  fetchWeatherData(location);
  clearMarkers();

  try {
    const openNowChecked = document.getElementById('open-now-checkbox').checked;

    let apiUrl = `/api/nearby?lat=${location.lat}&lng=${location.lng}&type=${currentPlaceType}`;
    if (keyword) apiUrl += `&keyword=${encodeURIComponent(keyword)}`;
    if (openNowChecked) apiUrl += '&opennow=true';

    const response = await fetch(apiUrl);
    const data = await response.json();

    if (data.status !== 'OK' || !data.results) {
      document.getElementById('places-container').innerHTML = `<div class="alert alert-info">No results found.</div>`;
      hideLoading();
      return;
    }

    const MIN_RATING = currentPlaceType === 'restaurant' ? 4.0 : 3.8;
    const MIN_REVIEWS = currentPlaceType === 'lodging' ? 8 : (currentPlaceType === 'restaurant' ? 20 : 10);
    const UNWANTED_TYPES = ["gas_station", "car_repair", "car_dealer", "convenience_store", "grocery_or_supermarket", "liquor_store", "market", "store"];

    const getDistanceInMeters = (lat1, lon1, lat2, lon2) => {
      const R = 6371000;
      const dLat = (lat2 - lat1) * Math.PI / 180;
      const dLon = (lon2 - lon1) * Math.PI / 180;
      const a = Math.sin(dLat/2)**2 + Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * Math.sin(dLon/2)**2;
      return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
    };

    const filtered = data.results.filter(place => {
      if (!place.geometry || !place.geometry.location) return false;
      const pLat = place.geometry.location.lat;
      const pLng = place.geometry.location.lng;
      const distance = getDistanceInMeters(location.lat, location.lng, pLat, pLng);
      if (distance > radius) return false;

      if (place.types && place.types.some(type => UNWANTED_TYPES.includes(type))) return false;
      if (place.rating && place.rating < MIN_RATING) return false;
      return true;
    });

    const sorted = filtered.sort((a, b) => {
      const aSig = a.user_ratings_total >= MIN_REVIEWS;
      const bSig = b.user_ratings_total >= MIN_REVIEWS;
      if (aSig && bSig) return b.rating - a.rating;
      if (aSig) return -1;
      if (bSig) return 1;
      return (b.user_ratings_total || 0) - (a.user_ratings_total || 0);
    });

    const container = document.getElementById('places-container');
    container.innerHTML = '';

    if (sorted.length === 0) {
      container.innerHTML = `<div class="alert alert-info">No high-rated places within ${radius}m.</div>`;
    } else {
      sorted.forEach((place, index) => {
        const card = createPlaceCard(place, index);
        container.appendChild(card);
        addMarker(place, index);
      });
    }

  } catch (err) {
    console.error('Error loading places:', err);
    document.getElementById('places-container').innerHTML = `<div class="alert alert-danger">Failed to load places.</div>`;
  } finally {
    hideLoading();
  }
}
