function searchNearbyOnHover(location) {
  if (document.getElementById('place-modal').classList.contains('show')) return;
  if (hoverTimeout) clearTimeout(hoverTimeout);

  hoverTimeout = setTimeout(() => {
    if (!hoverInfoWindow) {
      hoverInfoWindow = new google.maps.InfoWindow({
        disableAutoPan: true,
        maxWidth: 320,
        content: '<div class="p-2"><div class="spinner-border spinner-border-sm text-primary" role="status"></div> Finding nearby places...</div>'
      });
      google.maps.event.addListener(map, 'click', () => hoverInfoWindow.close());
      google.maps.event.addListener(hoverInfoWindow, 'closeclick', () => clearTimeout(hoverTimeout));
    } else {
      hoverInfoWindow.close();
    }

    hoverInfoWindow.setContent('<div class="p-2"><div class="spinner-border spinner-border-sm text-primary" role="status"></div> Finding nearby places...</div>');
    hoverInfoWindow.setPosition(location);
    hoverInfoWindow.open(map);

    const placesService = new google.maps.places.PlacesService(map);
    placesService.nearbySearch({
      location: location,
      radius: 300,
      type: currentPlaceType,
      rankBy: google.maps.places.RankBy.PROMINENCE
    }, (results, status) => {
      if (status !== google.maps.places.PlacesServiceStatus.OK || !results.length) {
        hoverInfoWindow.setContent(`<div class="p-2">No ${formatPlaceType(currentPlaceType)}s found here</div>`);
        return;
      }

      const MIN_RATING = currentPlaceType === 'restaurant' ? 4.0 : 3.8;
      const UNWANTED_TYPES = ["gas_station", "car_repair", "car_dealer", "convenience_store", "grocery_or_supermarket", "liquor_store", "market", "store"];

      const getDistanceInMeters = (lat1, lon1, lat2, lon2) => {
        const R = 6371000;
        const dLat = (lat2 - lat1) * Math.PI / 180;
        const dLon = (lon2 - lon1) * Math.PI / 180;
        const a = Math.sin(dLat/2)**2 + Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * Math.sin(dLon/2)**2;
        return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
      };

      const filtered = results.filter(place => {
        if (!place.geometry || !place.geometry.location) return false;
        const pLat = place.geometry.location.lat;
        const pLng = place.geometry.location.lng;
        const dist = getDistanceInMeters(location.lat, location.lng, pLat, pLng);
        if (dist > searchRadius) return false;

        if (place.types && place.types.some(type => UNWANTED_TYPES.includes(type))) return false;
        if (place.rating && place.rating < MIN_RATING) return false;
        return true;
      });

      const topPlaces = filtered.sort((a, b) => (b.rating || 0) - (a.rating || 0)).slice(0, 5);
      if (!topPlaces.length) {
        hoverInfoWindow.setContent(`<div class="p-2">No high-rated ${formatPlaceType(currentPlaceType)}s nearby</div>`);
        return;
      }

      let content = `<div class="p-2" style="max-width: 300px;">
        <h6 class="mb-2">Nearby ${formatPlaceType(currentPlaceType)}s</h6>
        <ul class="list-group list-group-flush ps-0">`;

      topPlaces.forEach(place => {
        const rating = place.rating ? `<span class="text-warning">${'★'.repeat(Math.round(place.rating))}</span>` : '<span class="text-muted">No rating</span>';
        const photoUrl = (place.photos?.[0]?.photo_reference)
          ? `/api/photo?photoreference=${place.photos[0].photo_reference}&maxwidth=100`
          : '';
        const icon = photoUrl ? `<img src="${photoUrl}" class="float-start me-2 rounded" style="width: 40px; height: 40px; object-fit: cover;">` : '';
        content += `
          <li class="list-group-item p-2 mb-1 border-0" style="cursor: pointer" onclick="showPlaceDetails('${place.place_id}'); if(hoverInfoWindow) hoverInfoWindow.close();">
            <div class="d-flex align-items-center">${icon}
              <div><strong>${place.name}</strong><br>${rating} ${place.user_ratings_total ? `<small>(${place.user_ratings_total})</small>` : ''}</div>
            </div>
          </li>`;
      });

      content += '</ul><div class="text-center"><small class="text-muted">Click to view details</small></div></div>';
      hoverInfoWindow.setContent(content);
    });
  }, 500);
}
