async function loadNearbyPlaces(location, keyword = '', radius = 1500) {
  showLoading(); clearMarkers();
  const openNow = document.getElementById('open-now-checkbox').checked;
  const service = new google.maps.places.PlacesService(map);

  let rawPlaces = [];
  if (keyword === 'dessert') {
    // 1) Gather dessert‐type places
    const searches = [
      { type: 'ice_cream_parlor' },
      { type: 'bakery' },
      { type: 'cafe', keyword: 'dessert' },
      { type: 'restaurant', keyword: 'dessert' },
      { type: 'restaurant', keyword: 'cake' }
    ];
    const seen = new Map();
    for (let s of searches) {
      const results = await new Promise(res =>
        service.nearbySearch({
          location, radius,
          type: s.type,
          keyword: s.keyword || '',
          openNow
        }, (r, st) => res(st === google.maps.places.PlacesServiceStatus.OK ? r : []))
      );
      results.forEach(p => seen.set(p.place_id, p));
      if (seen.size >= 50) break;
    }
    rawPlaces = Array.from(seen.values());
  } else {
    // non‐dessert categories...
    rawPlaces = await new Promise(res =>
      service.nearbySearch({ location, radius, type: keyword || currentPlaceType, openNow },
        (r, st) => res(st === google.maps.places.PlacesServiceStatus.OK ? r : []))
    );
  }

  // 2) Quality + distance filter
  let filtered = filterAndSortPlaces(rawPlaces, location, {
    radius,
    minRating: 3.5,
    minReviews: 5,
    unwantedTypes: ['gas_station','convenience_store']
  });

  // 3) If we're in dessert mode, further require positive dessert mentions
  if (keyword === 'dessert') {
    const dessertKeywords = ['dessert','cake','tiramisu','pastry','sweet','delicious','amazing dessert','gelato','ice cream'];
    const goodDesserts = [];
    // Only check the top 20 to save API calls
    for (let place of filtered.slice(0, 20)) {
      const details = await new Promise(res =>
        service.getDetails({ placeId: place.place_id, fields: ['reviews'] },
          (d, st) => res(st === google.maps.places.PlacesServiceStatus.OK ? d : null))
      );
      if (!details || !details.reviews) continue;
      const count = details.reviews
        .slice(0, 5) // top 5 reviews
        .filter(r => {
          const text = (r.text||'').toLowerCase();
          return dessertKeywords.some(k => text.includes(k));
        }).length;
      if (count >= 2) {
        goodDesserts.push(place);
      }
    }
    filtered = goodDesserts;
  }

  // 4) Render final list
  renderPlaces(filtered);
  hideLoading();
}
